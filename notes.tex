    \expandafter\ifpatchable\expandafter{\csname\@cmd\endcsname}{]}{%
    \expandafter\patchcmd\expandafter%
      {\csname\@cmd\endcsname}{]}{,.. this/.try]}{}{}%
    \pgfutil@namelet{\@cmd @orig}{\@cmd}%
    \expandafter\patchvenncmd\expandafter{\@cmd}}{}}}
